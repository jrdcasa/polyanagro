class Chain_Statistics(pag.Calculations):

    # #########################################################################
    def __init__(self, trj, top, dt=1, stride=1,removetmpfiles_ee=True, removetmpfiles_rg=True):

        super().__init__(trj, top, dt, stride)
        self._removetmpfiles_ee = removetmpfiles_ee
        self._removetmpfiles_rg = removetmpfiles_rg
        self._rg2_frame = {}
        self._ree2_frame = {}
        self._uree_frame = None
        self._rEE_ACF = None

        self._ree_max = sys.float_info.min
        self._ree_min = sys.float_info.max
        self._rg_max = sys.float_info.min
        self._rg_min = sys.float_info.max

    # #########################################################################
    def calculate(self, write_rg=True, pbc =True, diroutput="./", listendtoend=None, acfE2E=False, distributions=False):

        # Write messages
        LogMessage('info',"")
        LogMessage('info',"Calculating Radius of gyration...")
        start_time = datetime.datetime.now()

        LogMessage('info',"")
        LogMessage('info',"Calculating Radius of gyration...")
        if write_rg:
            LogMessage('info',"Radius of gyration written to {0:s} ".format(diroutput+"Rg.dat"))
        if not listendtoend is None:
            LogMessage('info',"")
            LogMessage('info',"Calculating End to End distance...")
            LogMessage('info',"End to End distance written to {0:s} \n".format(diroutput+"Ree.dat"))

        # Start ndarray to store the end to end vectors for autocorrelation function
        if acfE2E:
            nchains = len(self._nmols_array)
            nframes = self._trajectory.get_numframes()
            self._uree_frame = np.zeros([3, nchains, nframes], dtype=np.float32)
            self._rEE_ACF = np.zeros([nframes], dtype=np.float32)

        # Start calculations for each frame
        iframe_jumpfirst = 0
        for itrj in self._trajectory.trjlist:

            ini = 0 if iframe_jumpfirst == 0 else self._stride

            for iframe_local in range(ini, itrj.n_frames, self._stride):

                if iframe_local%self._freq == 0:
                    elapsed_time = datetime.datetime.now() - start_time
                    line = "IFRAME: {0:d} in {1:s} seconds".format\
                        (iframe_local, str(elapsed_time.total_seconds()))
                    LogMessage('info',"time: {0:s} seconds".format(line))

                if pbc:
                    self._unwrap_coordinates(itrj, iframe_local, pbc=pbc)

                self._single_rg(iframe_jumpfirst, filename=diroutput+"Rg.dat", tmpfileindiviualchain=distributions)
                self._single_ree(iframe_jumpfirst, filename=diroutput+"Ree.dat", listendtoend=listendtoend,tmpfileindiviualchain=distributions)
                iframe_jumpfirst += self._stride

        # Write final messages and time
        end_time = datetime.datetime.now()
        elapsed_time = end_time - start_time
        LogMessage('info',"TIME(Chain_Statistics): {0:s} seconds".format
             (str(elapsed_time.total_seconds())))

        if acfE2E:
            # Write messages
            start_time = datetime.datetime.now()
            LogMessage('info',"")
            LogMessage('info',"Calculating End-to-End autocorrelation vector...")

            self._acf_e2e(filename=diroutput+"rEE_ACF.dat")

            # Write final messages and time
            end_time = datetime.datetime.now()
            elapsed_time = end_time - start_time
            LogMessage('info',"TIME(End-to-End autocorrelation vector): {0:s} seconds".format
                 (str(elapsed_time.total_seconds())))

        if distributions:
            with open(".tmp_rg_distances.dat", 'a') as f:
                f.writelines("Rg_min_max: {0:.4f} {1:.4f}\n".format(self._rg_min, self._rg_max))
            if not listendtoend is None:
                with open(".tmp_ree_distances.dat", 'a') as f:
                    f.writelines("Ree_min_max: {0:.4f} {1:.4f}\n".format(self._ree_min, self._ree_max))

            self._distribution(".tmp_ree_distances.dat")
            self._distribution(".tmp_rg_distances.dat")

            print("Calculate distributions TO BE DONE!!!!!!")

    def _single_rg(self, iframe_total, filename="Rg.dat", write_result=True, ismassweight=False, tmpfileindiviualchain=False):

        mass = np.array(self._top._mass, dtype=np.float32)
        nchains = len(self._nmols_array)
        rgsq_ich_iframe = np.zeros([nchains, 4], dtype=np.float32)

        if ismassweight:
            calc_rg_openmp_massweigth(self._nmols_array, mass, self._coords_unwrap, rgsq_ich_iframe)
        else:
            calc_rg_openmp(self._nmols_array, mass, self._coords_unwrap, rgsq_ich_iframe)

        # Calculate the average
        rg2_avg = 0.0
        rg2_std = 0.0
        for ich in range(nchains):
            rg2_avg += rgsq_ich_iframe[ich,3]
        rg2_avg = rg2_avg / nchains
        # Calculate the standard deviation
        for ich in range(nchains):
            rg2_std += (rgsq_ich_iframe[ich,3] - rg2_avg)*(rgsq_ich_iframe[ich,3] - rg2_avg)
        rg2_std = np.sqrt(rg2_std/(nchains-1))

        self._rg2_frame[iframe_total*self._dt] = [rg2_avg, rg2_std]

        # Write a tmp file containing the radius of gyration for each chain.
        # Each line corresponds to all chains in the current frame
        # This is useful to calculate the radius of gyration distribution
        if tmpfileindiviualchain:
            try:
                if self._removetmpfiles_rg:
                    os.remove(".tmp_rg_distances.dat")
            except FileNotFoundError:
                pass
            with open(".tmp_rg_distances.dat",'a') as f:
                self._removetmpfiles_rg = False
                line = "{} ".format(iframe_total)
                for ich in range(nchains):
                    rg = rgsq_ich_iframe[ich,3]**0.5
                    line += " {0:.4f} ".format(rg)
                    self._rg_max = rg if rg > self._rg_max else self._rg_max
                    self._rg_min = rg if rg < self._rg_min else self._rg_min
                line += "\n"
                f.writelines(line)

        if write_result:
            i = iframe_total*self._dt
            if iframe_total == 0:
                with open(filename,'w') as f:
                    f.writelines("#   Time(ps)     Rg^2(A^2)     stdRg^2(A^2)\n")
                    f.writelines("#==========================================\n")

                    f.writelines("{0:>10.2f}   {1:>10.2f}   {2:>10.2f} \n".format(i,
                                                    self._rg2_frame[i][0],
                                                    self._rg2_frame[i][1]))
            else:
                with open(filename,'a') as f:
                    f.writelines("{0:>10.2f}   {1:>10.2f}   {2:>10.2f} \n".format(i,
                                                    self._rg2_frame[i][0],
                                                    self._rg2_frame[i][1]))

        return rg2_avg, rg2_std

def calculate(self, write_rg=True, pbc =True, diroutput="./", listendtoend=None, acfE2E=False, distributions=False):

    # Write messages
    LogMessage('info',"")
    LogMessage('info',"Calculating Radius of gyration...")
    start_time = datetime.datetime.now()

    LogMessage('info',"")
    LogMessage('info',"Calculating Radius of gyration...")
    if write_rg:
        LogMessage('info',"Radius of gyration written to {0:s} ".format(diroutput+"Rg.dat"))
    if not listendtoend is None:
        LogMessage('info',"")
        LogMessage('info',"Calculating End to End distance...")
        LogMessage('info',"End to End distance written to {0:s} \n".format(diroutput+"Ree.dat"))

    # Start ndarray to store the end to end vectors for autocorrelation function
    if acfE2E:
        nchains = len(self._nmols_array)
        nframes = self._trajectory.get_numframes()
        self._uree_frame = np.zeros([3, nchains, nframes], dtype=np.float32)
        self._rEE_ACF = np.zeros([nframes], dtype=np.float32)

    # Start calculations for each frame
    iframe_jumpfirst = 0
    for itrj in self._trajectory.trjlist:

        ini = 0 if iframe_jumpfirst == 0 else self._stride

        for iframe_local in range(ini, itrj.n_frames, self._stride):

            if iframe_local%self._freq == 0:
                elapsed_time = datetime.datetime.now() - start_time
                line = "IFRAME: {0:d} in {1:s} seconds".format\
                    (iframe_local, str(elapsed_time.total_seconds()))
                LogMessage('info',"time: {0:s} seconds".format(line))

            if pbc:
                self._unwrap_coordinates(itrj, iframe_local, pbc=pbc)

            self._single_rg(iframe_jumpfirst, filename=diroutput+"Rg.dat", tmpfileindiviualchain=distributions)
            self._single_ree(iframe_jumpfirst, filename=diroutput+"Ree.dat", listendtoend=listendtoend,tmpfileindiviualchain=distributions)
            iframe_jumpfirst += self._stride

    # Write final messages and time
    end_time = datetime.datetime.now()
    elapsed_time = end_time - start_time
    LogMessage('info',"TIME(Chain_Statistics): {0:s} seconds".format
         (str(elapsed_time.total_seconds())))

    if acfE2E:
        # Write messages
        start_time = datetime.datetime.now()
        LogMessage('info',"")
        LogMessage('info',"Calculating End-to-End autocorrelation vector...")

        self._acf_e2e(filename=diroutput+"rEE_ACF.dat")

        # Write final messages and time
        end_time = datetime.datetime.now()
        elapsed_time = end_time - start_time
        LogMessage('info',"TIME(End-to-End autocorrelation vector): {0:s} seconds".format
             (str(elapsed_time.total_seconds())))

    if distributions:
        with open(".tmp_rg_distances.dat", 'a') as f:
            f.writelines("Rg_min_max: {0:.4f} {1:.4f}\n".format(self._rg_min, self._rg_max))
        if not listendtoend is None:
            with open(".tmp_ree_distances.dat", 'a') as f:
                f.writelines("Ree_min_max: {0:.4f} {1:.4f}\n".format(self._ree_min, self._ree_max))

        self._distribution(".tmp_ree_distances.dat")
        self._distribution(".tmp_rg_distances.dat")

        print("Calculate distributions TO BE DONE!!!!!!")
